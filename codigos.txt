      function calculatePointsPerCell(config, comboCount, totalCells, collectedCount, gridPos, gridSize) { const bp = config.basePoints || 15; const mode = config.pointsPerCell; if (typeof mode === 'number') return mode; switch (mode) {
          case "random":
            return Math.floor(Math.random() * ((config.pointsRandomMax || 25) - (config.pointsRandomMin || 5) + 1)) + (config.pointsRandomMin || 5);
          case "increasing":
            return bp + collectedCount * 2;
          case "decreasing":
            return bp + (totalCells - collectedCount) * .5;
          case "combo":
            return Math.floor(bp * (1 + comboCount * (config.comboMultiplier || .1)));
          case "multiplier":
            return Math.floor(bp * (config.multiplier || 2));
          case "edge":
            if (gridPos.x === 0 || gridPos.x === gridSize - 1 || gridPos.y === 0 || gridPos.y === gridSize - 1) return Math.floor(bp * (config.edgeMultiplier || 2.5)); return bp;
          case "center": { const cx = Math.floor(gridSize / 2),
              cy = Math.floor(gridSize / 2); if (Math.abs(gridPos.x - cx) <= 1 && Math.abs(gridPos.y - cy) <= 1) return Math.floor(bp * (config.centerMultiplier || 2)); return bp; }
          case "lastStand":
            if (totalCells - collectedCount <= (config.lastStandThreshold || 3)) return Math.floor(bp * (config.lastStandMultiplier || 5)); return bp;
          default:
            return bp; } }
      
      function calculateCompletionBonus(config, totalCells, comboCount, elapsed) { const mode = config.completionBonus; const base = config.completionBonusPoints || 200; if (!mode || mode === 0) return 0; if (typeof mode === 'number') return mode; switch (mode) {
          case "perfect":
            return base;
          case "scale":
            return Math.floor(base * (config.completionBonusMultiplier || 1.5));
          case "speed": { const factor = config.completionSpeedFactor || .5; const tb = Math.max(0, base - Math.floor(elapsed / 1000) * 5); return tb + Math.floor(base * factor); }
          case "overkill":
            if (comboCount >= (config.overkillThreshold || 5)) return base * 3; return base;
          default:
            return base; } }

      function generatePowerUps() { powerUpsOnMap = []; if (wave < POWERUP_CONFIG.startWave) return; const candidates = getAvailablePowerUpTypes(); if (!candidates.length) return; const toSpawn = Math.min(POWERUP_CONFIG.spawnCount, candidates.length); for (let i = 0; i < toSpawn; i++) { const typeKey = candidates[Math.floor(Math.random() * candidates.length)]; let x, y, attempts = 0;

      // BUG#1 – timeSlow/freeze/doubleScore: no registerTimeout, handled in updateActiveEffects
      function applyNewPowerUpEffect(type) {
        switch (type) {
          case 'timeSlow':
            if (!originalSpeeds) { originalSpeeds = { normal: enemySpawnConfig.normal.speed, boss: enemySpawnConfig.boss.speed }; }
            enemySpawnConfig.normal.speed *= POWERUP_CONFIG.types.timeSlow.slowFactor;
            enemySpawnConfig.boss.speed *= POWERUP_CONFIG.types.timeSlow.slowFactor;
            break;
          case 'doubleScore':
            doublePointsActive = true;
            break;
          case 'freeze':
            enemies.forEach(e => e.speed *= .1);
            break;
          case 'teleporter': { const msgCells = messageCollectibles.length > 0 ? messageCollectibles : collectibles; if (msgCells.length > 0) { const target = msgCells[Math.floor(Math.random() * msgCells.length)];
              player.x = target.x;
              player.y = target.y;
              playerDrawX = player.x * CELL_SIZE + CELL_SIZE / 2;
              playerDrawY = player.y * CELL_SIZE + CELL_SIZE / 2;
              createExplosion(playerDrawX, playerDrawY, 30, '#00ffff', 1.5);
              SPECIAL_SOUNDS.teleport(); } break; }
          case 'starShower':
            for (let i = 0; i < 15; i++) collectibles.push({ x: Math.floor(Math.random() * GRID_SIZE), y: Math.floor(Math.random() * GRID_SIZE) });
            createExplosion(canvas.width / 2, canvas.height / 2, 40, '#ffdd00', 2);
            SPECIAL_SOUNDS.starPower();
            break;
          case 'clone': { const oldClone = activeEffects.find(e => e.type === 'clone'); if (oldClone && oldClone.intervalId) clearInterval(oldClone.intervalId);
            activeEffects = activeEffects.filter(e => e.type !== 'clone'); const cloneInterval = setInterval(() => { if (gameOver || !activeEffects.some(e => e.type === 'clone')) { clearInterval(cloneInterval); return; } const allCells = [...collectibles, ...messageCollectibles]; if (allCells.length > 0) { const target2 = allCells.reduce((a, b) => { const da = Math.abs(a.x - player.x) + Math.abs(a.y - player.y); const db = Math.abs(b.x - player.x) + Math.abs(b.y - player.y); return da < db ? a : b; });
                player.x = target2.x;
                player.y = target2.y;
                playerDrawX = player.x * CELL_SIZE + CELL_SIZE / 2;
                playerDrawY = player.y * CELL_SIZE + CELL_SIZE / 2;
                checkCollectibles(); } }, 500);
            activeEffects.push({ type: 'clone', expires: Date.now() + POWERUP_CONFIG.types.clone.duration, intervalId: cloneInterval }); break; }
        }
      }

      function updateActiveEffects() {
        const now = Date.now();
        for (let i = activeEffects.length - 1; i >= 0; i--) {
          if (now >= activeEffects[i].expires) {
            const ef = activeEffects[i];
            if (ef.type === 'timeSlow') { if (originalSpeeds) { enemySpawnConfig.normal.speed = originalSpeeds.normal;
                enemySpawnConfig.boss.speed = originalSpeeds.boss; } originalSpeeds = null; }
            if (ef.type === 'freeze') { enemies.forEach(e => e.speed = Math.min(.2, e.speed * 10)); }
            if (ef.type === 'doubleScore' && !activeEffects.some((e, ei) => ei !== i && e.type === 'doubleScore')) { doublePointsActive = false; }
            if (ef.type === 'clone' && ef.intervalId) { clearInterval(ef.intervalId); }
            activeEffects.splice(i, 1);
          }
        }
      }

      // ══ ZONE INDICATOR (new responsive animated version) ══
      let lastZoneColor = '#00ffcc'; // color de la última zona (para la salida)
let lastZoneText = ''; // texto de la última zona
      let zoneEffectActive = false; // ¿hay animación en curso?
let zoneEffectColor = '#00ffcc'; // color actual
let zoneEffectStartTime = 0; // cuándo empezó (para la expansión)
let zoneEffectOut = false; // true = estamos saliendo
let zoneEffectOutStart = 0; // cuándo empezó a salir
let zoneLastType = null; // para detectar cambios
function drawZoneIndicator() {
  if (gameOver) return;

  // ── Detectar zona actual ──
  let currentZoneColor = null;
  let currentText = '';

  if (isInTripleBonusZone())      { currentZoneColor = '#FFD700'; currentText = '||ZONA PUNTOS X3||'; }
  else if (isInDoubleBonusZone()) { currentZoneColor = '#ff00ff'; currentText = '||ZONA PUNTOS X2||'; }
  else if (isInBonusZone())       { currentZoneColor = '#ffcc00'; currentText = '||ZONA DE PUNTOS||'; }
  else if (isInSafeZone())        { currentZoneColor = '#00ffcc'; currentText = '||ZONA SEGURA||'; }

  // ── Lógica de entrada / salida ──
  if (currentZoneColor) {
    if (!zoneEffectActive || zoneEffectColor !== currentZoneColor) {
      zoneEffectActive = true;
      zoneEffectColor = currentZoneColor;
      lastZoneColor = currentZoneColor;
      lastZoneText = currentText;
      zoneEffectStartTime = Date.now();
      zoneEffectOut = false;
    } else if (zoneEffectOut) {
      zoneEffectOut = false;
      zoneEffectActive = true;
    }
    zoneLastType = currentZoneColor;
  } else {
    if (zoneEffectActive && !zoneEffectOut) {
      zoneEffectOut = true;
      zoneEffectOutStart = Date.now();
    }
  }

  if (!zoneEffectActive) return;

  const now = Date.now();
  const duration = 600; // ms
  let progress;

  if (!zoneEffectOut) {
    progress = Math.min(1, (now - zoneEffectStartTime) / duration);
  } else {
    progress = Math.max(0, 1 - (now - zoneEffectOutStart) / duration);
    if (progress <= 0) {
      zoneEffectActive = false;
      return;
    }
  }

  const activeColor = zoneEffectOut ? lastZoneColor : zoneEffectColor;
  const activeText  = zoneEffectOut ? lastZoneText  : currentText;

  // ═══════════════════════════════════════
  // 🌆 ONDA CYBERPUNK EXPANSIVA + DATOS CORRUPTOS EN EL BORDE
  // ═══════════════════════════════════════
  const px = playerDrawX, py = playerDrawY;
  const maxRadius = Math.sqrt(canvas.width**2 + canvas.height**2) * 0.5;
  const startRadius = CELL_SIZE * 0.2;
  const currentRadius = startRadius + (maxRadius - startRadius) * easeOutCubic(progress);

  if (progress > 0) {
    ctx.save();

    // Círculo base translúcido
    const gradient = ctx.createRadialGradient(px, py, currentRadius * 0.2, px, py, currentRadius);
    gradient.addColorStop(0, activeColor + '30');
    gradient.addColorStop(0.7, activeColor + '08');
    gradient.addColorStop(1, 'transparent');
    ctx.fillStyle = gradient;
    ctx.beginPath(); ctx.arc(px, py, currentRadius, 0, Math.PI * 2); ctx.fill();

    // Borde neón delgado
    ctx.strokeStyle = activeColor;
    ctx.lineWidth = 2;
    ctx.globalAlpha = 0.8 + Math.sin(now * 0.02) * 0.2;
    ctx.beginPath(); ctx.arc(px, py, currentRadius, 0, Math.PI * 2); ctx.stroke();

    // Anillos concéntricos internos (sutiles)
    for (let i = 1; i <= 3; i++) {
      const r = currentRadius * (0.6 + i * 0.12);
      ctx.beginPath(); ctx.arc(px, py, r, 0, Math.PI * 2);
      ctx.strokeStyle = activeColor;
      ctx.lineWidth = 1;
      ctx.globalAlpha = 0.25;
      ctx.stroke();
    }

    // Líneas de escaneo horizontal (más finas)
    for (let i = 0; i < 4; i++) {
      const yPos = (py - currentRadius + (i * currentRadius * 2) / 3 + now * 0.1) % (currentRadius * 2) + py - currentRadius;
      ctx.strokeStyle = activeColor;
      ctx.lineWidth = 0.8;
      ctx.globalAlpha = 0.2;
      ctx.beginPath(); ctx.moveTo(px - currentRadius, yPos); ctx.lineTo(px + currentRadius, yPos); ctx.stroke();
    }

    // 🔹 DATOS CORRUPTOS EN EL BORDE DE LA ONDA (efecto principal)
    const corruptionCount = 12 + Math.floor(progress * 10);
    ctx.fillStyle = activeColor;
    for (let i = 0; i < corruptionCount; i++) {
      const angle = Math.random() * Math.PI * 2;
      const r = currentRadius * (0.92 + Math.random() * 0.12);
      const cx = px + Math.cos(angle) * r;
      const cy = py + Math.sin(angle) * r;
      ctx.globalAlpha = 0.6 + Math.random() * 0.4;
      ctx.fillRect(cx - 1, cy - 1, 2 + Math.random() * 6, 2);
    }

    // Destellos verticales muy finos (solo unos pocos)
    ctx.strokeStyle = '#ffffff';
    ctx.lineWidth = 0.6;
    ctx.globalAlpha = 0.1 * progress;
    for (let i = 0; i < 4; i++) {
      const x = px - currentRadius + (i / 3) * currentRadius * 2;
      ctx.beginPath(); ctx.moveTo(x, py - currentRadius); ctx.lineTo(x, py + currentRadius); ctx.stroke();
    }

    // ⚡ LÍNEAS CINÉTICAS ALREDEDOR DEL JUGADOR (Sandevistan adaptado)
    const kineticCount = 1 + Math.floor(progress * 1);
    for (let i = 0; i < kineticCount; i++) {
      const angle = (i / kineticCount) * Math.PI * 2 + now * 0.003;
      const innerRadius = CELL_SIZE * 0.1;
      const outerRadius = innerRadius + CELL_SIZE * (0.5 + progress * 2);
      const x1 = px + Math.cos(angle) * innerRadius;
      const y1 = py + Math.sin(angle) * innerRadius;
      const x2 = px + Math.cos(angle) * outerRadius;
      const y2 = py + Math.sin(angle) * outerRadius;
      ctx.strokeStyle = activeColor;
      ctx.lineWidth = 0.5;
      ctx.globalAlpha = 0.35 + Math.sin(now * 0.02 + i) * 0.15;
      ctx.beginPath();
      ctx.moveTo(x1, y1);
      ctx.lineTo(x2, y2);
      ctx.stroke();
    }

    ctx.restore();
  }

  // ═══════════════════════════════════════
  // 🖥️ TEXTO HUD CYBERPUNK CENTRADO + BARRA DE CARGA
  // ═══════════════════════════════════════
  const textX = canvas.width / 2;
  const textY = canvas.height / 2;
  const distToText = Math.hypot(textX - px, textY - py);
  const revealMargin = CELL_SIZE * 2;
  let textReveal = Math.max(0, Math.min(1, (currentRadius - distToText + revealMargin) / (revealMargin * 2)));
  textReveal = easeOutCubic(textReveal);
  const textAlpha = progress * textReveal;

  if (textAlpha > 0.01) {
    const fontSize = Math.floor(canvas.width * 0.045);
    ctx.save();
    ctx.font = `bold ${fontSize}px "Courier New", monospace`;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';

    // Fondo de la ventana del HUD
    const textWidth = ctx.measureText(activeText).width + 20;
    const textHeight = fontSize * 1.5;
    const boxY = textY - textHeight/2;

    ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
    ctx.fillRect(textX - textWidth/2, boxY, textWidth, textHeight);
    ctx.strokeStyle = activeColor;
    ctx.lineWidth = 1;
    ctx.strokeRect(textX - textWidth/2, boxY, textWidth, textHeight);

    // Texto principal
    ctx.fillStyle = activeColor;
    ctx.globalAlpha = textAlpha;
    ctx.fillText(activeText, textX, textY);

    // Cursor parpadeante
    if (Math.floor(now / 500) % 2 === 0) {
      const cursorX = textX + textWidth/2 - 8;
      ctx.fillStyle = '#ffffff';
      ctx.globalAlpha = textAlpha;
      ctx.fillRect(cursorX, textY - fontSize*0.4, 6, fontSize*0.8);
    }

    // Glitch ocasional
    if (Math.random() < 0.1 && textAlpha > 0.5) {
      const offsetX = (Math.random() - 0.5) * 4;
      const offsetY = (Math.random() - 0.5) * 4;
      ctx.fillStyle = activeColor;
      ctx.globalAlpha = textAlpha * 0.50;
      ctx.fillText(activeText, textX + offsetX, textY + offsetY);
    }

    // ▓▓ BARRA DE CARGA ▓▓
    const barWidth = textWidth * 1;
    const barHeight = 4;
    const barY2 = boxY + textHeight + 1.5;

    // Fondo de la barra
    ctx.fillStyle = 'rgba(0, 0, 0, 0.7)';
    ctx.fillRect(textX - barWidth/2, barY2 - barHeight/2, barWidth, barHeight);
    ctx.strokeStyle = activeColor;
    ctx.lineWidth = 0.8;
    ctx.strokeRect(textX - barWidth/2, barY2 - barHeight/2, barWidth, barHeight);

    // Relleno de la barra según progress
    const fillWidth = barWidth * progress;
    ctx.fillStyle = activeColor;
    ctx.fillRect(textX - barWidth/2, barY2 - barHeight/2, fillWidth, barHeight);

    ctx.restore();
  }

      function checkEnemyCollision() {
        if (isInSafeZone() || Date.now() < invulnerableUntil) return false;
        for (let e of enemies) {
          const cells = e.getOccupiedCells();
          for (let cell of cells) {
            if (cell.x < 0 || cell.x >= GRID_SIZE || cell.y < 0 || cell.y >= GRID_SIZE) continue;
            if (player.x === cell.x && player.y === cell.y) {
              // 1) SHIELD first
              const shieldIdx = activeEffects.findIndex(ef => ef.type === 'shield');
              if (shieldIdx !== -1) {
                stopMotor();
                activeEffects.splice(shieldIdx, 1);
                const idx = enemies.indexOf(e);
                if (idx !== -1) { createExplosion(e.x * CELL_SIZE + CELL_SIZE / 2, e.y * CELL_SIZE + CELL_SIZE / 2, 25, '#aaddff');
                  enemies.splice(idx, 1); }
                createFloatingText(playerDrawX, playerDrawY - 20, "ESCUDO ROTO", true);
                SPECIAL_SOUNDS.shieldBreak();
                invulnerableUntil = Date.now() + 800;
                // Shake on shield break
                targetTiltX = (Math.random() - .5) * .8;
                targetTiltZ = (Math.random() - .5) * .8;
                return false;
              }
              // 2) REVIVE second
              if (hasReviveTotem) {
                stopMotor();
                hasReviveTotem = false;
                // BUG#2 – restore speeds on revive
                if (activeEffects.some(ef => ef.type === 'timeSlow')) { if (originalSpeeds) { enemySpawnConfig.normal.speed = originalSpeeds.normal;
                    enemySpawnConfig.boss.speed = originalSpeeds.boss; } originalSpeeds = null;
                  activeEffects = activeEffects.filter(ef => ef.type !== 'timeSlow'); }
                if (activeEffects.some(ef => ef.type === 'freeze')) { enemies.forEach(e2 => e2.speed = Math.min(.2, e2.speed * 10));
                  activeEffects = activeEffects.filter(ef => ef.type !== 'freeze'); }
                enemies = [];
                // BUG#A – cancel any active move animation on revive
                if (moveAnimId) { cancelAnimationFrame(moveAnimId);
                  moveAnimId = null;
                  moving = false; }
                createExplosion(playerDrawX, playerDrawY, 50, '#88ff88', 2.2);
                createFloatingText(canvas.width / 2, canvas.height / 2 - 50, "REVIVIDO", true, true);
                playerGlow = 200;
                invulnerableUntil = Date.now() + 1500;
                SPECIAL_SOUNDS.reviveReady();
                return false;
              }
              // 3) GAME OVER
              // BUG#6 – cancel clone interval before game over
              for (let ef of activeEffects) { if (ef.type === 'clone' && ef.intervalId) clearInterval(ef.intervalId); }
              // BUG#A – cancel RAF
              if (moveAnimId) { cancelAnimationFrame(moveAnimId);
                moveAnimId = null; }
              gameOver = true;
              gameOverTime = Date.now();
              stopMotor();
              createExplosion(player.x * CELL_SIZE + CELL_SIZE / 2, player.y * CELL_SIZE + CELL_SIZE / 2, 50, '#ff0000', 1.5);
              playHitSound();
              return true;
            }
          }
        }
        return false;
      }

// Función actualizada con recogida forzada después de 1.5 segundos
function updateAttracted() {
  const tx = playerDrawX, ty = playerDrawY;
  const now = Date.now();

  for (let i = attractedPool.length - 1; i >= 0; i--) {
    const a = attractedPool[i];

    // ⏱️ Si lleva más de 1.5 segundos en el pool, lo traemos directamente al jugador
    if (now - a.startTime > 1500) {
      // Forzamos la recogida inmediata
      a.drawX = tx;
      a.drawY = ty;
      // Ahora lo recogeremos porque la distancia es 0
    }

    const dx = tx - a.drawX;
    const dy = ty - a.drawY;
    const dist = Math.sqrt(dx * dx + dy * dy);

    // Radio de recogida ampliado (si falla, se capturará por tiempo)
    if (dist < CELL_SIZE * 0.8) {
      // Recoger este objeto atraído
      if (a.isMessage && a.data) {
        const totalCells = messageCollectibles.length + attractedPool.filter(x => x.isMessage).length;
        const collectedCount = totalCells - messageCollectibles.length - attractedPool.filter(x => x.isMessage).length + 1;
        let pts = calculatePointsPerCell(currentWaveConfig, comboCount, totalCells, collectedCount, { x: Math.round(a.drawX / CELL_SIZE), y: Math.round(a.drawY / CELL_SIZE) }, GRID_SIZE);
        if (doublePointsActive) pts *= 2;
        score += pts;
        scoreSinceLastDoubleBonus += pts;
        premiumCollectedSinceLastTripleBonus++;
        createExplosion(a.drawX, a.drawY, 12, '#ff88ff');
        createFloatingText(a.drawX, a.drawY, pts, true);
        playPremiumCollect();
        comboCount++;
        if (score > highScore) { highScore = score; localStorage.setItem('odparHighScore', highScore); }
        updateUI();

        const remMsg = messageCollectibles.length + attractedPool.filter(x => x.isMessage).length - 1;
        if (remMsg === 0 && currentWaveMessage) {
          const elapsed = Date.now() - waveStartTime;
          let bonus = calculateCompletionBonus(currentWaveConfig, totalCells, comboCount, elapsed);
          if (doublePointsActive) bonus *= 2;
          if (currentWaveConfig.bonusOnCompleteEnabled !== false && bonus > 0) {
            score += bonus;
            if (score > highScore) { highScore = score; localStorage.setItem('odparHighScore', highScore); }
            createExplosion(canvas.width / 2, canvas.height / 2, 50, '#ffd700', 2.5);
            floatingTexts.push(new FloatingText(canvas.width / 2, canvas.height / 2 - 40, `MSG COMPLETO! +${bonus}`, '#FFD700', true, true, true));
            playTripleBonusCollect();
            updateUI();
          }
        }
      } else {
        let pts = doublePointsActive ? 16 : 8;
        score += pts;
        scoreSinceLastDoubleBonus += pts;
        createExplosion(a.drawX, a.drawY, 8, '#ffd700');
        createFloatingText(a.drawX, a.drawY, pts);
        playNormalCollect();
        if (score > highScore) { highScore = score; localStorage.setItem('odparHighScore', highScore); }
        updateUI();
        const remNorm = collectibles.length + attractedPool.filter(x => !x.isMessage).length - 1;
        if (remNorm === 0) completeWave();
      }
      attractedPool.splice(i, 1);
      continue;
    }

    // Física normal (si no lo hemos forzado)
    const force = Math.min(2.5, 0.5 + 800 / (dist * dist + 1));
    a.vx += (dx / dist) * force;
    a.vy += (dy / dist) * force;
    const maxSpd = CELL_SIZE * 0.5;   // velocidad máxima en píxeles/frame
    const spd = Math.sqrt(a.vx * a.vx + a.vy * a.vy);
    if (spd > maxSpd) {
      a.vx = (a.vx / spd) * maxSpd;
      a.vy = (a.vy / spd) * maxSpd;
    }
    a.drawX += a.vx;
    a.drawY += a.vy;
  }
}

      function executeMove(dir) {
        let newX = player.x,
          newY = player.y;
        switch (dir) {
          case 'up':
            newY--; break;
          case 'down':
            newY++; break;
          case 'left':
            newX--; break;
          case 'right':
            newX++; break;
          case 'up-left':
            newX--;
            newY--; break;
          case 'up-right':
            newX++;
            newY--; break;
          case 'down-left':
            newX--;
            newY++; break;
          case 'down-right':
            newX++;
            newY++; break; }
        if (newX < 0 || newX >= GRID_SIZE || newY < 0 || newY >= GRID_SIZE) return false;
        moving = true;
        let startX = playerDrawX,
          startY = playerDrawY,

        const speedActive = activeEffects.some(e => e.type === 'speed');
        let duration = Math.max(MIN_MOVE_DELAY, speedActive ? currentMoveDelay / 2 : currentMoveDelay);
        let animStart = performance.now();
        
        function easeInOutCubic(t) { return t < .5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2; }
        
        function animateMove(now) {
          if (gameOver) { moving = false;
            moveAnimId = null; return; }
          let elapsed = now - animStart,
            t = Math.min(1, elapsed / duration),
            ease = easeInOutCubic(t);
          playerDrawX = startX + (endX - startX) * ease;
          playerDrawY = startY + (endY - startY) * ease;
          if (t < 1) { moveAnimId = requestAnimationFrame(animateMove); }
          else { moveAnimId = null;
            playerDrawX = endX;
            playerDrawY = endY;
            moving = false;
            lastMoveTimestamp = performance.now(); if (currentMoveDelay > MIN_MOVE_DELAY) currentMoveDelay = Math.max(MIN_MOVE_DELAY, currentMoveDelay * .9);
            startNextMove(); }
        }
        moveAnimId = requestAnimationFrame(animateMove);
        playMoveSound();
        startMotor();
        const tiltMap = { up: [-.3, 0], down: [.3, 0], left: [0, -.4], right: [0, .4], 'up-left': [-.2, -.3], 'up-right': [-.2, .3], 'down-left': [.2, -.3], 'down-right': [.2, .3] };

        targetAngle = angleMap[dir] || 0;

  if (progress > 0.88 && progress < 0.95) {
