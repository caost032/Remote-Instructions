function applyProsody(text,style,pauseMult=1.0){
  const prof=VOICE_PROFILES[style]||VOICE_PROFILES['reflection'];
  const parts=[];
  const re=/([^.!?;:,…]+[.!?;:,…]?)/g;
  let m;
  while((m=re.exec(text))!==null){
    const seg=m[1].trim();if(!seg)continue;
    const last=seg[seg.length-1];
    let pause=0;
    if(/[.!?…]/.test(last))pause=Math.round(prof.pauseDot*pauseMult);
    else if(/[;:,]/.test(last))pause=Math.round(prof.pauseComma*pauseMult);
    parts.push({text:seg,pauseAfterMs:pause});
  }
  if(!parts.length)parts.push({text:text.trim(),pauseAfterMs:Math.round(prof.pauseDot*pauseMult)});
  return{segments:parts,utteranceConfig:{rate:prof.rate,pitch:prof.pitch,volume:prof.volume}};
}

/* ════════════════════════════════════════════════════════════════
   INGENIERO E — WORD ALIGNER v3 — NO REGRESIÓN
   Mantiene posición acumulada para evitar resaltar palabras anteriores
   ════════════════════════════════════════════════════════════════ */
class WordAligner {
  constructor(){
    this._lastCharIdx = -1; // v3: posición acumulada
    this._totalChars = 0;
  }
  reset(){
    this._lastCharIdx = -1;
    this._totalChars = 0;
  }
  /**
   * v3: align con no-regresión garantizada
   * Usa charIndex cuando está disponible, y fallback calibrado
   * La clave: nunca resaltar una posición <= _lastCharIdx
   */
  align(utt,text,rate,onWord){
    this._lastCharIdx = -1;
    this._totalChars = text.length;
    const words=text.split(/\s+/).filter(Boolean);
    const timers=[];
    let _boundaryFired=false;
    const self=this;

    if(utt){
      utt.addEventListener('boundary',e=>{
        if(e.name!=='word'||e.charIndex==null)return;
        _boundaryFired=true;
        timers.forEach(clearTimeout);timers.length=0;
        // v3: solo avanzar si charIndex > lastCharIdx
        if(e.charIndex <= self._lastCharIdx) return;
        self._lastCharIdx = e.charIndex;
        const word=text.substring(e.charIndex,e.charIndex+(e.charLength||8));
        try{onWord(word,e.charIndex)}catch{}
      });
    }

    // Fallback: timing predictivo con no-regresión
    const fallbackTO=setTimeout(()=>{
      if(_boundaryFired)return;
      const BASE_MS_PER_CHAR=62;
      const msPerChar=BASE_MS_PER_CHAR*(0.88/Math.max(0.4,rate));
      let offset=0;
      let cumChars=0;
      words.forEach((word,i)=>{
        const charPos=cumChars;
        const t=setTimeout(()=>{
          // v3: no-regresión en fallback también
          if(charPos <= self._lastCharIdx){cumChars+=word.length+1;return}
          self._lastCharIdx=charPos;
          try{onWord(word,charPos)}catch{}
        },offset);
        timers.push(t);
        offset+=word.length*msPerChar+(msPerChar*0.75);
        cumChars+=word.length+1;
      });
    },350);
    timers.push(fallbackTO);
    return{cancel(){timers.forEach(clearTimeout)}};
  }
}

    this._lang=lang;
    const pick=()=>{this._prefVoice=VoiceEnhancer.getBestFemale(lang)};
    if(window.speechSynthesis){window.speechSynthesis.onvoiceschanged=pick;pick()}
  }
  setBackend(name){this.backend=name;store.voiceBackend=name}

  async speak(text,style,{onStart,onEnd,onWord}={}){
    if(!text||text.trim()==='·'){if(onEnd)setTimeout(onEnd,350);return}
    if(style==='silence'||style==='epitaph'||style==='requiem'){
      if(onEnd)setTimeout(onEnd,style==='epitaph'||style==='requiem'?1800:700);return}
    this.stop();
    this.active=true;this.paused=false;this._advancePending=false;
    store.voiceActive=true;store.voicePaused=false;
    this.onWordCallback=onWord||null;
    this._onEndCallback=onEnd||null;
    this.aligner.reset(); // v3: resetear posición acumulada
    setVoiceActive(true);updateNVUI();

    const {segments,utteranceConfig}=applyProsody(text,style,this._userPauseMult);
    const rate=clamp(utteranceConfig.rate*this._userRate,0.4,2.0);
    const pitch=clamp(utteranceConfig.pitch*this._userPitch,0.5,2.0);
    const volume=clamp(utteranceConfig.volume*this._userVolume,0,1);
    this._currentRate=rate;

    if(onStart)onStart();

    if(this.backend==='piper'&&this._piperStrategy?.initialized){
      await this._speakPiper(text,rate,onEnd);
    }else{
      await this._speakMultiUtterance(segments,rate,pitch,volume);
    }
  }

  async _speakMultiUtterance(segments,rate,pitch,volume){
    if(!window.speechSynthesis){this._finish();return}
    this._segments=segments;this._segmentIdx=0;
    this._speakNextSegment(rate,pitch,volume);
  }

  _speakNextSegment(rate,pitch,volume){
    if(!this.active||this._segmentIdx>=this._segments.length){this._finish();return}
    const seg=this._segments[this._segmentIdx];
    const utt=new SpeechSynthesisUtterance(seg.text);
    utt.lang=this._lang;utt.rate=rate;utt.pitch=pitch;utt.volume=volume;
    if(this._prefVoice)utt.voice=this._prefVoice;

    if(this.onWordCallback){
      this._alignerHandle?.cancel();
      this._alignerHandle=this.aligner.align(utt,seg.text,rate,(word,charIdx)=>{
        if(this.active&&this.onWordCallback)this.onWordCallback(word,charIdx);
      });
    }

    utt.onstart=()=>{store.voiceActive=true;updateNVUI()};
    utt.onend=()=>{
      if(!this.active)return;
      this._alignerHandle?.cancel();this._alignerHandle=null;
      this._segmentIdx++;
      const pause=seg.pauseAfterMs||0;
      clearTimeout(this._ttsTimeout);
      if(pause>40&&this._segmentIdx<this._segments.length){
        this._ttsTimeout=setTimeout(()=>{if(this.active&&!this.paused)this._speakNextSegment(rate,pitch,volume)},pause);
      }else{this._speakNextSegment(rate,pitch,volume)}
    };
    utt.onerror=e=>{if(e.error!=='interrupted')this._finish()};

    const safeMs=seg.text.length*75+1800;
    clearTimeout(this._ttsTimeout);
    this._ttsTimeout=setTimeout(()=>{if(this.active&&this._segmentIdx<this._segments.length)this._finish()},safeMs);

    window.speechSynthesis.speak(utt);
  }

  async _speakPiper(text,rate,onEnd){
    try{
      if(!this._piperStrategy?.initialized){const prof=applyProsody(text,'reflection',this._userPauseMult);await this._speakMultiUtterance(prof.segments,rate,1.0,this._userVolume);return}
      const buf=await this._piperStrategy.synthesize(text);
      if(!audioCtx){this._finish();return}
      const src=audioCtx.createBufferSource();src.buffer=buf;src.playbackRate.value=rate;
      src.connect(AN.voiceGain);src.onended=()=>this._finish();src.start();this._piperSource=src;
    }catch(e){console.warn('[NV10 Piper]',e.message);this._finish()}
  }

  pause(){
    if(!this.active)return;
    if(this.backend==='local'&&window.speechSynthesis?.speaking)window.speechSynthesis.pause();
    clearTimeout(this._ttsTimeout);this.paused=true;store.voicePaused=true;updateNVUI();announce('Pausa');
  }
  resume(){
    if(!this.paused)return;
    if(this.backend==='local'&&window.speechSynthesis?.paused)window.speechSynthesis.resume();
    this.paused=false;store.voicePaused=false;updateNVUI();announce('Reanudando');
  }
  stop(){
    this._alignerHandle?.cancel();this._alignerHandle=null;
    this.aligner.reset();
    clearTimeout(this._ttsTimeout);
    if(window.speechSynthesis&&(window.speechSynthesis.speaking||window.speechSynthesis.pending))window.speechSynthesis.cancel();
    if(this._piperSource){try{this._piperSource.stop()}catch{}this._piperSource=null}
    this._segments=[];this._segmentIdx=0;this._advancePending=false;
    this._finish(true);
  }
  _finish(silent=false){
    this._alignerHandle?.cancel();this._alignerHandle=null;
    this.aligner.reset();
    clearTimeout(this._ttsTimeout);
    this.active=false;this.paused=false;this._advancePending=false;
    store.voiceActive=false;store.voicePaused=false;
    setVoiceActive(false);clearHL();updateNVUI();
    if(!silent&&this._onEndCallback){
      const cb=this._onEndCallback;this._onEndCallback=null;
      setTimeout(cb,280);
    }
  }
  markAdvancePending(){this._advancePending=true}
  setUserParams({rate,pitch,volume,pauseMult}){
    if(rate!==undefined)this._userRate=rate;if(pitch!==undefined)this._userPitch=pitch;
    if(volume!==undefined)this._userVolume=volume;if(pauseMult!==undefined)this._userPauseMult=pauseMult;
    store.voiceRate=this._userRate;store.voicePitch=this._userPitch;store.voiceVol=this._userVolume;store.voicePauseMult=this._userPauseMult;
    setVoiceVolume(this._userVolume);
  }
  async activatePiper(modelPath){
    this._piperStrategy=new PiperVoiceStrategy(modelPath);
    await this._piperStrategy.initialize();
    if(this._piperStrategy.initialized){this.setBackend('piper');return true}
    return false;
  }
}

}

/**
 * INGENIERO E — speakBlock v3 — Resaltado preciso SIN REGRESIÓN
 * Busca la palabra desde la posición charIdx en adelante.
 * Nunca retrocede en el texto.
 */
function speakBlock(blockEl,style,lang='es-ES',onEnd=null){
  if(!blockEl)return;
  const clone=blockEl.cloneNode(true);
  clone.querySelectorAll('.word-hl').forEach(e=>e.replaceWith(document.createTextNode(e.textContent)));
  const text=clone.textContent.trim();
  const p=blockEl.querySelector('p');
  let _lastHighlightPos=-1; // v3: posición de último resaltado

  window.narrativeVoice._lang=lang;
  window.narrativeVoice.speak(text,style,{
    onEnd(){
      window.narrativeVoice.markAdvancePending();
      clearHL();
      _lastHighlightPos=-1;
      if(onEnd)onEnd();
    },
    onWord(word,charIdx){
      if(!p||!word||word.length<2)return;
      // v3: obtener el texto plano actual del párrafo para referencia
      // Limpiar highlights anteriores preservando texto
      clearHL();
      if(!p||!p.isConnected)return;

      const cleanWord=word.replace(/[^a-záéíóúüñA-ZÁÉÍÓÚÜÑ\w]/g,'').toLowerCase();
      if(!cleanWord||cleanWord.length<2)return;

      try{
        // v3: buscar DESDE _lastHighlightPos en adelante en el texto completo
        const fullText=p.textContent;
        // Construir regex que busca desde después de la última posición
        const searchFrom=Math.max(0,_lastHighlightPos);
        const textAhead=fullText.substring(searchFrom);
        // Buscar la palabra sin importar capitalización pero respetando límites
        const re=new RegExp(`\\b${cleanWord.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')}[a-záéíóúüñ]*\\b`,'i');
        const match=re.exec(textAhead);
        if(!match)return;
        const absoluteIdx=searchFrom+match.index;
        const matchedWord=match[0];

        // v3: actualizar posición acumulada
        _lastHighlightPos=absoluteIdx+matchedWord.length;

        // Navegar el árbol DOM para encontrar el nodo de texto en absoluteIdx
        const walker=document.createTreeWalker(p,NodeFilter.SHOW_TEXT,null,false);
        let node=walker.nextNode();
        let cumLen=0;
        while(node){
          const nodeLen=node.textContent.length;
          if(cumLen+nodeLen>absoluteIdx){
            // La palabra está en este nodo
            const localIdx=absoluteIdx-cumLen;
            if(localIdx<0||localIdx+matchedWord.length>nodeLen){node=walker.nextNode();cumLen+=nodeLen;continue}
            const bef=node.textContent.substring(0,localIdx);
            const aft=node.textContent.substring(localIdx+matchedWord.length);
            const span=document.createElement('span');
            span.className='word-hl';span.textContent=matchedWord;
            const frag=document.createDocumentFragment();
            frag.appendChild(document.createTextNode(bef));
            frag.appendChild(span);
            frag.appendChild(document.createTextNode(aft));
            node.replaceWith(frag);
            try{span.scrollIntoView({block:'nearest',behavior:'smooth'})}catch{}
            break;
          }
          cumLen+=nodeLen;node=walker.nextNode();
        }
      }catch(e){/* silenciar errores de DOM */}
    }
  });
}

/* ════════════════════════════════════════════════════════════════
   INGENIERO G — NLU v3: 32 comandos + wake-word + EJECUCIÓN FIABLE
   v3: los comandos ahora ejecutan usando _nluExec que garantiza
   que la función se llama en el contexto correcto del módulo
   ════════════════════════════════════════════════════════════════ */
let recognition=null,voiceOn=false,srRetries=0,_wakeWordMode=true,_cmdTimeout=null;

function toggleVoice(){
  const SR=window.SpeechRecognition||window.webkitSpeechRecognition;
  if(!SR){alert('Reconocimiento de voz no disponible en este navegador.');return}
  voiceOn=!voiceOn;
  if(voiceOn){srRetries=0;$('c-voice').classList.add('rec');$('v-ind').classList.add('on');_wakeWordMode=true;_startSR()}
  else{$('c-voice').classList.remove('rec');$('v-ind').classList.remove('on');_stopSR()}
}

function _startSR(){
  if(recognition)_stopSR();
  const SR=window.SpeechRecognition||window.webkitSpeechRecognition;
  recognition=new SR();recognition.lang='es-ES';recognition.continuous=true;recognition.interimResults=false;recognition.maxAlternatives=3;
  recognition.onresult=e=>{
    // v3: recopilar TODOS los resultados del evento, incluyendo alternativas
    const transcripts=[];
    for(let i=e.resultIndex;i<e.results.length;i++){
      for(let j=0;j<e.results[i].length;j++){
        transcripts.push(e.results[i][j].transcript.toLowerCase().trim());
      }
    }
    const cmd=transcripts.join(' ');
    if(_wakeWordMode){
      if(/n[uú]cleo|escucha|oye|hey/.test(cmd)){
        _wakeWordMode=false;
        announce('Escuchando comando…');
        const span=$('v-ind').querySelector('span:last-child');
        if(span)span.textContent='Comando…';
        clearTimeout(_cmdTimeout);
        _cmdTimeout=setTimeout(()=>{
          _wakeWordMode=true;
          const s=$('v-ind').querySelector('span:last-child');
          if(s)s.textContent='Escuchando…';
        },7000);
      }
      return;
    }
    // v3: ejecutar con cada transcripción hasta que alguna funcione
    let executed=false;
    for(const t of transcripts){
      if(_nluExec(t)){executed=true;break}
    }
    if(!executed)_nluExec(cmd); // fallback al joined
    _wakeWordMode=true;
    clearTimeout(_cmdTimeout);
    const s=$('v-ind').querySelector('span:last-child');
    if(s)s.textContent='Escuchando…';
  };
  recognition.onerror=e=>{
    if(e.error==='not-allowed'){voiceOn=false;$('c-voice').classList.remove('rec');$('v-ind').classList.remove('on');return}
    if(voiceOn&&srRetries<6){srRetries++;setTimeout(()=>{if(voiceOn)_startSR()},1200)}
  };
  recognition.onend=()=>{if(voiceOn)setTimeout(()=>{if(voiceOn)_startSR()},200)};
  try{recognition.start()}catch{}
}
function _stopSR(){if(recognition){recognition.onend=null;recognition.onerror=null;try{recognition.abort()}catch{}recognition=null}}

/**
 * v3: _nluExec — retorna true si ejecutó un comando, false si no
 * Todas las funciones se llaman directamente (mismo módulo ES)
 */
function _nluExec(cmd){
  if(!cmd)return false;
  // Navegación
  if(/siguiente|adelante|continu[aá]|avanza|pr[oó]ximo/.test(cmd)){nextForce();return true}
  if(/anterior|atr[aá]s|retrocede|vuelve|regresa/.test(cmd)){prev();return true}
  if(/inicio|portada|principio|empezar|comienzo/.test(cmd)){goTo(0);return true}
  if(/final|dedicator|[uú]ltimo/.test(cmd)){goTo(store.N-1);return true}
  // Mapa
  if(/mapa|ver mapa|abrir mapa/.test(cmd)){toggleMap($('c-map'));return true}
  // Lectura
  if(/leer todo|lectura continua|autoavanzar/.test(cmd)){initAudio();toggleReadAll();return true}
  if(/modo frase|frase a frase/.test(cmd)){togglePhraseMode();return true}
  if(/pausa|pausar/.test(cmd)){togglePauseTTS();return true}
  if(/reanudar|contin[uú]a leyendo/.test(cmd)){const nv=window.narrativeVoice;if(nv.paused)nv.resume();return true}
  if(/lee|habla|voz alta|narrar|leer esto/.test(cmd)){initAudio();speakCurrentBlock();return true}
  if(/silencio|calla|para|detener voz/.test(cmd)){stopTTS();return true}
  // Exportar
  if(/exportar|guardar|export/.test(cmd)){exportJSON();return true}
  if(/markdown/.test(cmd)){exportMarkdown();return true}
  // Tema
  if(/oscuro|modo noche|modo oscuro/.test(cmd)){setDark(true);return true}
  if(/claro|modo d[ií]a|modo claro/.test(cmd)){setDark(false);return true}
  if(/tema|dise[nñ]o|galer[ií]a/.test(cmd)){openThemeGallery();return true}
  // Velocidad de voz
  if(/m[aá]s r[aá]pido|acelera|m[aá]s veloz/.test(cmd)){
    const nv=window.narrativeVoice;nv._userRate=clamp(nv._userRate+0.12,0.4,2.0);store.voiceRate=nv._userRate;_syncVoicePanel();announce(`Velocidad: ${nv._userRate.toFixed(1)}`);return true}
  if(/m[aá]s lento|desacelera|m[aá]s despacio/.test(cmd)){
    const nv=window.narrativeVoice;nv._userRate=clamp(nv._userRate-0.12,0.4,2.0);store.voiceRate=nv._userRate;_syncVoicePanel();announce(`Velocidad: ${nv._userRate.toFixed(1)}`);return true}
  if(/m[aá]s aguda|sube tono|tono alto/.test(cmd)){
    const nv=window.narrativeVoice;nv._userPitch=clamp(nv._userPitch+0.1,0.5,2.0);store.voicePitch=nv._userPitch;_syncVoicePanel();return true}
  if(/m[aá]s grave|baja tono|tono bajo/.test(cmd)){
    const nv=window.narrativeVoice;nv._userPitch=clamp(nv._userPitch-0.1,0.5,2.0);store.voicePitch=nv._userPitch;_syncVoicePanel();return true}
  if(/cambiar voz|siguiente voz|otra voz/.test(cmd)){_rotateVoice();return true}
  // Volumen con número
  const volMatch=cmd.match(/volumen\s+(\d+)/);
  if(volMatch){const n=parseInt(volMatch[1]);if(!isNaN(n)){const v=clamp(n/100,0,1);window.narrativeVoice._userVolume=v;store.voiceVol=v;setVoiceVolume(v);_syncVoicePanel();return true}}
  // Inmersión
  if(/inmersi[oó]n|modo inmersi[oó]n|pantalla limpia/.test(cmd)){toggleImmersion();return true}
  if(/quitar inmersi[oó]n|salir inmersi[oó]n/.test(cmd)){toggleImmersion(false);return true}
  // Panel de voz
  if(/configurar voz|panel voz|ajustar voz/.test(cmd)){initAudio();openVoicePanel();return true}
  // Contenido
  if(/cargar obra|nueva obra|abrir obra/.test(cmd)){openContentPanel();return true}
  // Ir a sección por nombre o número
  const secMatch=cmd.match(/(?:ir a|ve a|secci[oó]n|cap[ií]tulo)\s+(.+)/);
  if(secMatch){
    const query=secMatch[1].trim();
    const found=store.slides.findIndex((s,i)=>i>0&&(s.number?.toLowerCase().includes(query)||s.title?.toLowerCase().includes(query)));
    if(found>=0){goTo(found);return true}
  }
  return false;
}

function _rotateVoice(){
  const ranked=VoiceEnhancer.getRankedVoices(window.narrativeVoice._lang);
  if(!ranked.length)return;
  const cur=ranked.findIndex(r=>r.voice===window.narrativeVoice._prefVoice);
  const next=ranked[(cur+1)%ranked.length];
  window.narrativeVoice._prefVoice=next.voice;
  announce(`Voz: ${next.voice.name}`);_syncVoicePanel();
}

function exportJSON(){

function exportMarkdown(){
  let md=`# ${store.meta.title||'Experiencia'}\n\n`;
  md+=`*Diario de lectura — ${new Date().toLocaleDateString('es-ES',{year:'numeric',month:'long',day:'numeric'})}*\n\n---\n\n`;
  let lastS=-1;
  for(const e of store.journal){
    const slide=store.slides[e.s]||{};const block=slide.blocks?.[e.b];
    const text=(block?.text||'').replace(/<[^>]+>/g,'');const style=block?.style||'n';
    if(e.s!==lastS){md+=`\n## ${slide.number||slide.title||'Sección'}: ${slide.title||''}\n\n`;lastS=e.s}
    if(style==='silence'||style==='epitaph'||style==='requiem')continue;
    const lbl=STYLE_LABELS[style];const prefix=lbl?`**${lbl}:** `:'';
    const dur=e.d>1000?`${(e.d/1000).toFixed(1)}s`:`${e.d}ms`;
    md+=`> ${prefix}${text}\n> *(${dur})*\n\n`;
  }
  md+=`---\n\n*Progreso: ${Math.round((store.globalBlk()/store.totalBlks)*100)}% · Bloques leídos: ${store.journal.length}/${store.totalBlks}*\n`;
  _dl(md,`${store.config.exportTitle||'nucleo'}-${new Date().toISOString().slice(0,10)}.md`,'text/markdown');
  announce('Diario exportado como Markdown');
}
function _dl(content,filename,type){

  const a=document.createElement('a');a.href=url;a.download=filename;a.click();URL.revokeObjectURL(url);
}

  $('theme-gallery').addEventListener('click',e=>{if(e.target===$('theme-gallery'))closeThemeGallery()});
  $('theme-gallery').addEventListener('keydown',e=>{if(e.key==='Escape')closeThemeGallery()});
  document.querySelectorAll('.tg-card').forEach(card=>{
    card.addEventListener('click',()=>applyAppTheme(card.dataset.theme));
    card.addEventListener('keydown',e=>{if(e.key==='Enter'||e.key===' '){e.preventDefault();applyAppTheme(card.dataset.theme)}});
  });
  $('tg-url-btn').addEventListener('click',async()=>{
    const url=$('tg-url-input').value.trim();if(!url)return;
    try{await loadExternalTheme(url);document.documentElement.setAttribute('data-app-theme','custom')}
    catch(e){alert('Error cargando tema: '+e.message)}
  });
  $('cp-close').addEventListener('click',closeContentPanel);
  $('content-panel').addEventListener('click',e=>{if(e.target===$('content-panel'))closeContentPanel()});
  $('content-panel').addEventListener('keydown',e=>{if(e.key==='Escape')closeContentPanel()});
  $('cp-load-btn').addEventListener('click',()=>loadAndApply($('cp-url-input').value.trim()||null));
  document.querySelectorAll('.cp-preset').forEach(btn=>{
    btn.addEventListener('click',()=>{const url=btn.dataset.url;if(!url){closeContentPanel();return}$('cp-url-input').value=url;loadAndApply(url)});
  });

  // Panel de voz
  bindVoicePanelEvents();

  /* ── Teclado ── */
  document.addEventListener('keydown',e=>{
    resetIdle();
    // En modo inmersión: cualquier tecla llama peek
    if(store.immersionMode){immersionPeek()}
    const tag=(e.target||{}).tagName;if(tag==='INPUT'||tag==='TEXTAREA')return;
    switch(e.key){
      case 'ArrowRight':case ' ':
        e.preventDefault();
        if(store.phraseMode)advancePhrase();else nextForce();break;
      case 'ArrowLeft':e.preventDefault();prev();break;
      case 'Escape':
        if(store.immersionMode){toggleImmersion(false)}
        else if(mapOpen)toggleMap($('c-map'));
        else if($('theme-gallery').classList.contains('on'))closeThemeGallery();
        else if($('content-panel').classList.contains('on'))closeContentPanel();
        else if($('voice-panel').classList.contains('on'))closeVoicePanel();
        else if(store.phraseMode){e.preventDefault();disablePhraseMode()}
        else if(window.narrativeVoice.active){e.preventDefault();stopTTS()}
        else{e.preventDefault();goTo(0)}break;
      case 'Home':e.preventDefault();goTo(0);break;
      case 'End':e.preventDefault();goTo(store.N-1);break;
      case 'm':case 'M':e.preventDefault();toggleMap($('c-map'));break;
      case 'd':case 'D':if(!e.ctrlKey&&!e.metaKey){e.preventDefault();openThemeGallery()}break;
      case 'o':case 'O':if(!e.ctrlKey&&!e.metaKey){e.preventDefault();openContentPanel()}break;
      case 'n':case 'N':if(!e.ctrlKey&&!e.metaKey){e.preventDefault();initAudio();openVoicePanel()}break;
      case 'i':case 'I':if(!e.ctrlKey&&!e.metaKey){e.preventDefault();toggleImmersion()}break;
      case 'p':case 'P':
        e.preventDefault();
        if(window.narrativeVoice.active||window.narrativeVoice.paused)togglePauseTTS();
        break;
      case 'l':case 'L':
        if(e.shiftKey){e.preventDefault();initAudio();toggleReadAll()}
        else{e.preventDefault();initAudio();const nv=window.narrativeVoice;if(nv.paused)nv.resume();else if(nv.active)stopTTS();else speakCurrentBlock()}break;
      case 'f':case 'F':e.preventDefault();if(store.phraseMode)advancePhrase();else togglePhraseMode();break;
      case 'v':case 'V':e.preventDefault();toggleVoice();break;
      case 't':case 'T':if(!e.ctrlKey&&!e.metaKey){e.preventDefault();toggleDarkLight()}break;
      case 's':case 'S':
        if(e.ctrlKey&&e.shiftKey){e.preventDefault();exportMarkdown()}
        else if(e.ctrlKey||e.metaKey){e.preventDefault();exportJSON()}break;
    }
  });

  /* ── Touch swipe ── */
  let tx=0,ty=0,sw=0,pinch0=0;
  document.addEventListener('touchstart',e=>{
    if(e.touches.length===2){pinch0=Math.hypot(e.touches[0].clientX-e.touches[1].clientX,e.touches[0].clientY-e.touches[1].clientY)}
    else{tx=e.touches[0].clientX;ty=e.touches[0].clientY;sw=1}
    resetIdle();
    // Inmersión: toque llama peek
    if(store.immersionMode)immersionPeek();
  },{passive:true});
  document.addEventListener('touchmove',e=>{
    if(e.touches.length===2){
      const d=Math.hypot(e.touches[0].clientX-e.touches[1].clientX,e.touches[0].clientY-e.touches[1].clientY);
      if(pinch0>0&&d>pinch0*1.5&&!mapOpen){toggleMap($('c-map'));pinch0=0}
    }
    if(!sw)return;
    const dx=e.touches[0].clientX-tx,dy=e.touches[0].clientY-ty;
    if(Math.abs(dx)>8||Math.abs(dy)>8)e.preventDefault();
  },{passive:false});
  document.addEventListener('touchend',e=>{
    if(!sw)return;sw=0;
    if(e.target.closest('button,.dot,.ctrl'))return;
    // En inmersión solo hace peek, no navega
    if(store.immersionMode){immersionPeek();return}
    const dx=e.changedTouches[0].clientX-tx,dy=e.changedTouches[0].clientY-ty;
    const adx=Math.abs(dx),ady=Math.abs(dy);
    if(Math.max(adx,ady)<20)return;
    if(adx>ady){if(adx>34)dx<0?nextForce():prev()}
    else{if(ady>48)dy>0?nextForce():prev()}
  },{passive:true});

  /* ── Wheel ── */
  let wLock=0;
  document.addEventListener('wheel',e=>{
    if(mapOpen)return;e.preventDefault();if(wLock)return;wLock=1;setTimeout(()=>wLock=0,360);
    (e.deltaY>0||e.deltaX>0)?nextForce():prev();resetIdle();
  },{passive:false});

  /* ── Click zonas ── */
  document.addEventListener('click',e=>{
    if(mapOpen||e.target.closest('button,.dot,.ctrl,#theme-gallery,#content-panel,#voice-panel'))return;
    if(store.immersionMode){immersionPeek();return}
    const x=e.clientX,now=Date.now();
    if((window.narrativeVoice.active||window.narrativeVoice.paused)&&now-lastClickTime<380){togglePauseTTS();lastClickTime=0;return}
    lastClickTime=now;
    if(store.phraseMode){advancePhrase();return}
    if(x<innerWidth*0.3)prev();else if(x>innerWidth*0.7)nextForce();
    resetIdle();
  });

  /* ── Pointermove (spotlight + parallax) ── */
  const cvSpot=$('cv-spotlight');let movePending=false;
  document.addEventListener('pointermove',e=>{
    if(isTouch()||movePending)return;movePending=true;
    requestAnimationFrame(()=>{
      movePending=false;
      if(cvSpot){
        cvSpot.style.background=`radial-gradient(circle 340px at ${e.clientX}px ${e.clientY}px,transparent 0%,rgba(0,0,0,0.055) 100%)`;
        cvSpot.classList.add('on');clearTimeout(spotTO);spotTO=setTimeout(()=>cvSpot.classList.remove('on'),2200);
      }
      setPan(e.clientX);setLightParallax(e.clientX-innerWidth/2,e.clientY-innerHeight/2);resetIdle();
    });
  },{passive:true});

  /* ── Giroscopio ── */
  function reqOriPerm(){
    if(typeof DeviceOrientationEvent!=='undefined'&&typeof DeviceOrientationEvent.requestPermission==='function'){
      DeviceOrientationEvent.requestPermission().then(s=>{if(s==='granted')orientActive=true}).catch(()=>{});
    }else{orientActive=true}
  }
  window.addEventListener('deviceorientation',e=>{
    if(!orientActive||e.gamma==null)return;
    const now=Date.now();if(now-orientThrottle<100)return;orientThrottle=now;
    const targetFwt=Math.round(320+clamp(e.gamma,-30,30)*3.2);
    if(Math.abs(targetFwt-currentFwt)>5){
      currentFwt=lerp(currentFwt,targetFwt,0.14);
      document.documentElement.style.setProperty('--fwt',Math.round(currentFwt));
    }
  },{passive:true});
  document.addEventListener('click',function initOri(){reqOriPerm();document.removeEventListener('click',initOri)},{once:true});

  /* ── ResizeObserver (canvases + landscape) ── */
  let resizeTO=null;
  const ro=new ResizeObserver(()=>{
    clearTimeout(resizeTO);
    resizeTO=setTimeout(()=>{
      const w=innerWidth,h=innerHeight;
      cvP.width=w;cvP.height=h;cvL.width=w;cvL.height=h;_cv2.width=w;_cv2.height=h;
      if(_boidSystem){_boidSystem.canvas.width=w;_boidSystem.canvas.height=h}
      if(mapOpen)renderMap();
    },100);
  });
  ro.observe(document.body);

  /* ── Fullscreen change ── */
  document.addEventListener('fullscreenchange',()=>{
    if(!document.fullscreenElement&&store.immersionMode){
      // El usuario salió de fullscreen manualmente: desactivar inmersión
      store.immersionMode=false;
      document.documentElement.classList.remove('immersion-mode');
      $('c-immersion').classList.remove('immersed','lit');
    }
  });

  /* ── Sensor luz ambiental (cuando disponible) ── */
  if('AmbientLightSensor' in window){
    try{
      navigator.permissions.query({name:'ambient-light-sensor'}).then(perm=>{
        if(perm.state==='denied')return;
        try{
          const sensor=new AmbientLightSensor();let lastAct=0;
          sensor.onreading=()=>{
            const lux=sensor.illuminance,dark=document.documentElement.dataset.theme==='dark',now=Date.now();
            if(now-lastAct<5000)return;
            if(lux<3&&!dark){toggleDarkLight();lastAct=now}
            else if(lux>70&&dark){toggleDarkLight();lastAct=now}
          };
          sensor.onerror=()=>{};sensor.start();
        }catch{}
      }).catch(()=>{});
    }catch{}
  }
}

(async function main(){
  const _t0=performance.now();

  /* ── Preferencias guardadas (SINCRONO — antes de render) ── */
  try{const sd=localStorage.getItem('nv10-dark');if(sd==='dark'||sd==='light')document.documentElement.dataset.theme=sd}catch{}
  const savedTheme=getSavedTheme('warm-paper');
  store.currentAppTheme=savedTheme;
  const isDark=document.documentElement.dataset.theme==='dark';
  applyThemeById(store.currentAppTheme,isDark);
  $('ico-sun').style.display=isDark?'none':'';$('ico-moon').style.display=isDark?'':'none';

  /* ── Restaurar modo inmersión ── */
  try{if(localStorage.getItem('nv10-immersion')==='1')toggleImmersion(true)}catch{}

  /* ── Splash: visual mínimo mientras carga ── */
  setEmo('epiphany');

  /* ── IA atmosférica ── */
  initAIWorker();

  /* ── Sistemas visuales — lazy si low power ── */
  if(!prefersReducedMotion()){
    startAmb();
    startEmoLight();
    if(_boidSystem)_boidSystem.setParams({style:'epiphany'});
  }

  /* ── Loop reactividad voz ── */
  requestAnimationFrame(voiceReactivityLoop);

  /* ── Eventos (bind early para respuesta inmediata) ── */
  bindEvents();

  /* ── Cargar contenido ── */
  const params=new URLSearchParams(window.location.search);
  const contentUrl=params.get('content');
  let content;
  const ls=$('loading-screen');
  try{
    if(contentUrl){
      ls.querySelector('.splash-by').textContent='Cargando obra…';
      content=await loadContentFromURL(contentUrl);
    }else{
      content=getFallback();
    }
  }catch(e){console.warn('[NV10] Usando fallback:',e);content=getFallback()}

  /* ── Aplicar tema del libro si no hay preferencia guardada ── */
  try{
    const ut=localStorage.getItem('nv10-app-theme');
    if(!ut&&content.config.defaultTheme){
      store.currentAppTheme=content.config.defaultTheme;
      applyThemeById(store.currentAppTheme,isDark);
    }
  }catch{}

  /* ── Inicializar voz ── */
  initVoice(content.config.ttsLang||'es-ES');

  /* ── Service Worker ── */
  if('serviceWorker' in navigator)navigator.serviceWorker.register('./sw.js').catch(()=>{});

  /* ── Inicializar con contenido ── */
  await initWithContent(content,true);

  /* ── Salida del splash — mínimo 1.6s (más rápido que v9) ── */
  const SPLASH_MIN=1600;
  const elapsed=performance.now()-_t0;
  const remaining=Math.max(0,SPLASH_MIN-elapsed);
  setTimeout(()=>{
    setEmo('n');
    if(_boidSystem)_boidSystem.setParams({style:'n'});
    ls.classList.add('hidden');
    setTimeout(()=>{ls.style.display='none'},600);
    // Medir FPS tras splash
    scheduleIdle(()=>adaptToFPS(_boidSystem));
  },remaining);

  console.info('%c✦ NÚCLEO VIVO v10.0 ✦','color:#c88552;font-size:15px;font-weight:bold;font-style:italic');
  console.info('%cUltra Alta Fidelidad — 15 Temas (6 nuevos) · 38 Estilos · Landscape Perfecto','color:#8a7e72;font-size:11px');
  console.info('Teclado: I=Inmersión · N=Panel voz · L=Leer · Shift+L=Todo · V=Voz · F=Frase · M=Mapa · D=Diseño');
  console.info('Comandos de voz: di "núcleo" y luego el comando (ej: "núcleo siguiente", "núcleo leer todo", "núcleo inmersión")');
  console.info(`Cargado en ${(performance.now()-_t0).toFixed(0)}ms`);
})();
